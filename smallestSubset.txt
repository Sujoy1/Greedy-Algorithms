#include <bits/stdc++.h>
using namespace std;

int minElements(int arr[], int n)
{
    int sum = 0 , sum1=0;
    
    for(int i=0;i<n;i++)
    sum += arr[i];
    
    sort(arr,arr+n);
    
    for(int i=0;i<n;i++)
    {
        sum1 += arr[n-1-i];
        if(sum1>sum-sum1)
        return (i+1);
    }
    
    
}

int main() {
	
	int arr[] = {3, 1, 7, 1};
    int n = sizeof(arr) / sizeof(arr[0]);
    cout << minElements(arr, n) << endl;
    
	return 0;
}
