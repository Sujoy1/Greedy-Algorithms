long long int maximizeSum(long long int a[], int n, int k)
    {
        // Your code goes here
        long long int sum =0;
        
        sort(a,a+n);
        
        for(int i=0;i<n;i++)
        {
            if(a[i]<0&&k>0)
            {
                a[i] = -a[i];
                k--;
            }
            sum +=a[i];
        }
        
        return sum;
    }
