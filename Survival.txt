#include <bits/stdc++.h>
using namespace std;

void survival(int S, int N , int M)
{
    if(N<M)
    cout<<"not possible";
    
    int food  = S*M;
    if(food%N==0)
    cout<<food/N<<endl;
    else
    cout<<(food/N)+1<<endl;
    
}

int main()
{
    int S = 10, N = 16, M = 2;
    survival(S, N, M);
    return 0;
}
