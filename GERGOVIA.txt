#include <bits/stdc++.h>
using namespace std;
 
int main() {
	// your code goes here
	long long int n,i,cost,bottles;
	while(true)
	{
	    cin>>n;
	    if(n==0)
	    break;
	    else
	    {
	        long long int arr[n];
	    
	        for(int i=0;i<n;i++)
	        {
	            cin>>arr[i];
	            
	        }
	        cost=0;
	        bottles=arr[0];
	        for(i=1;i<n;i++)
	        {
	            cost += abs(bottles);
	            bottles += arr[i];
	        }
	        cout<<cost<<endl;
	        
	    }
	}
	return 0;
}
