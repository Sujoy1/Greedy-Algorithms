#include <iostream>
using namespace std;

int maxSum(int s1[],int s2[],int s3[],int n1,int n2,int n3)
{
    
    int sum1 = 0 ;int sum2 = 0 ;int sum3 =0;
    int max_sum,min_sum;
    for(int i=0;i<n1;i++)
    sum1 += s1[i];
    for(int i=0;i<n2;i++)
    sum2 += s2[i];
    for(int i=0;i<n3;i++)
    sum3 += s3[i];
    
    int i=0,j=0,k=0;
    
    while(i<n1&&j<n2&&k<n3)
    {
        if(sum1==sum2&&sum2==sum3)
        return sum1;
        else if(sum1==sum2)
        {
            while(sum3>sum1)
            {
                sum3 -= s3[k];
                k++;
                if(sum3==sum1)
                return sum1;
                
            }
        }
        else if(sum3==sum2)
        {
            while(sum1>sum2)
            {
                sum1 -= s1[i];
                i++;
                if(sum3==sum1)
                return sum1;
                
            }
        }
        else if(sum1==sum3)
        {
            while(sum2>sum1)
            {
                sum2 -= s2[j];
                j++;
                if(sum2==sum1)
                return sum1;
                
            }
        }
        else
        {
            max_sum  = max(sum1,max(sum2,sum3));
            min_sum  = min(sum1,min(sum2,sum3));
            if(max_sum==sum1){
            while(max_sum>min_sum)
            {
                max_sum -= s1[i];
                i++;
            }
            sum1 = max_sum;
            }
            else if(max_sum==sum2){
            while(max_sum>min_sum)
            {
                max_sum -= s2[j];
                j++;
            }
            sum2 = max_sum;
            }
            else {
            while(max_sum>min_sum)
            {
                max_sum -= s3[k];
                k++;
            }
            sum3 = max_sum;
            }
        }
    }
    return 0;
    
}

int main()
{
    int stack1[] = { 3, 2, 1, 1, 1 };
    int stack2[] = { 4, 3, 2 };
    int stack3[] = { 1, 1, 4, 1 };
 
    int n1 = sizeof(stack1) / sizeof(stack1[0]);
    int n2 = sizeof(stack2) / sizeof(stack2[0]);
    int n3 = sizeof(stack3) / sizeof(stack3[0]);
 
    cout << maxSum(stack1, stack2, stack3, n1, n2, n3)
         << endl;
    return 0;
}
